# Commands to push your Houseiana project to GitHub

# Step 1: Add GitHub as remote (replace YOUR_USERNAME with your actual GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/houseiana.git

# Step 2: Set main branch
git branch -M main

# Step 3: Push to GitHub
git push -u origin main

# If you get authentication errors, you may need to use a Personal Access Token
# Go to GitHub Settings → Developer Settings → Personal Access Tokens → Generate new token
# Then use: git push https://YOUR_TOKEN@github.com/YOUR_USERNAME/houseiana.git main
